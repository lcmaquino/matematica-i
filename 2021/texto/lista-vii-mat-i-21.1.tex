\begin{center}
  \textbf{Lista VII}
\end{center}

\begin{enumerate}
  \item Sejam as funções dadas por $f(x) = 4x + 3$ e $g(x) = x^2 + 2x - 3$.
  Determine as raízes da função dada por $g\circ f(x)$.
 
  \item Dadas as funções definidas por $f(x) = 2x - 8$ e $g(x) = ax + b$, 
  se $f\circ g(x) = 12x + 8$, então calcule o valor de $a + b$.

  \item Se $f(x) = ax + 2$ e $f\circ f(1) = 1$, então determine $a$.

  \item Considerando que $f(x) = -4x + 3$ e $g^{-1}(x) = \dfrac{x - m}{3}$,
   determine o valor de $m$ sabendo que $f\circ g(x) = g\circ f(x)$.

  \item Sejam as funções dadas por $f(x) = 2x + 1$ e $g(x) = x - 1$. Classifique
  as alternativas abaixo em verdadeiro ou falso.
  
  \begin{enumerate}
    \item $f\circ g(x) = g\circ f(x)$
    \item $g\circ f(2) = 5$
    \item $f\circ g(1) = 1$
    \item $f\circ f(1) + g\circ g(1) = 6$
  \end{enumerate}
  
\end{enumerate}
